    Verbalizer¶

    wordbase.verbalizer ¶

    Verbalizer: mapping between labels and surface form strings.

    A Verbalizer maps each label T to one or more surface form strings and provides a reverse lookup from surface form back to label. It is used by LLMDistribution to translate between the model's textual outputs and the typed labels used by application code.

    Typical usage example:

    from wordbase.verbalizer import Verbalizer, BOOL_VERBALIZER

v = Verbalizer({True: ["yes", "correct"], False: ["no", "incorrect"]})
print(v.label_for("yes"))     # True
print(v.label_for("YES"))     # True  (case-insensitive by default)
print(v.label_for("maybe"))   # None

print(BOOL_VERBALIZER.label_for("true"))  # True

    Verbalizer ¶

    Verbalizer(mapping: dict[T, list[str]], case_sensitive: bool = False)

    Maps labels to surface form strings with optional case folding.

    Each label may have one or more surface forms. A reverse lookup table is built at construction so that label_for() runs in O(1). Duplicate surface forms across different labels raise AmbiguousVerbalizerError at construction time so callers discover conflicts early.

    def __init__(
    self,
    mapping: dict[T, list[str]],
    case_sensitive: bool = False,
) -> None:
    """Construct a Verbalizer from a label-to-surfaces mapping.

    Args:
        mapping: Dict mapping each label to a list of surface form
            strings.  Each surface form must appear under exactly one
            label.
        case_sensitive: When False (default) surface forms are compared
            after lowercasing.  When True the comparison is exact.

    Raises:
        AmbiguousVerbalizerError: If any surface form appears under more
            than one label.
    """
    self._mapping: dict[T, list[str]] = {k: list(v) for k, v in mapping.items()}
    self._case_sensitive = case_sensitive
    self._reverse: dict[str, T] = {}

    for label, surfaces in self._mapping.items():
        for surface in surfaces:
            key = surface if case_sensitive else surface.lower()
            if key in self._reverse:
                if self._reverse[key] != label:
                    raise AmbiguousVerbalizerError(
                        f"Surface form {surface!r} is mapped to more than one "
                        f"label ({self._reverse[key]!r} and {label!r}). "
                        "Each surface form must appear under exactly one label."
                    )
            else:
                self._reverse[key] = label

    @classmethod
def identity(cls, labels: list[T]) -> Verbalizer[T]:
    """Create a Verbalizer where each label maps to its own string form.

    Args:
        labels: Ordered list of labels.  Each is converted to a string
            via str() and used as its sole surface form.

    Returns:
        A new Verbalizer with mapping {label: [str(label)]} for each label.
    """
    return cls({label: [str(label)] for label in labels})

    @classmethod
def from_enum(cls, enum_type: type[T]) -> Verbalizer[T]:
    """Create a Verbalizer from an Enum class.

    Each enum member maps to two surface forms: its value (as a string)
    and its name.

    Args:
        enum_type: An Enum subclass.  Members without a string-compatible
            value are represented by their name only.

    Returns:
        A new Verbalizer with mapping {member: [str(member.value),
        member.name]} for each member.
    """
    mapping: dict[T, list[str]] = {}
    for member in enum_type:  # type: ignore[attr-defined]
        surfaces = []
        value_str = str(member.value)
        name_str = member.name
        surfaces.append(value_str)
        if name_str != value_str:
            surfaces.append(name_str)
        mapping[member] = surfaces
    return cls(mapping)

    def all_surfaces(self) -> list[str]:
    """Return a flat list of all surface forms across all labels.

    Returns:
        List of surface form strings in label-insertion order.
    """
    result: list[str] = []
    for surfaces in self._mapping.values():
        result.extend(surfaces)
    return result

    def label_for(self, surface: str) -> T | None:
    """Look up the label for a surface form string.

    Args:
        surface: The surface form to look up.

    Returns:
        The label associated with this surface form, or None if the
        surface form is not recognised.
    """
    key = surface if self._case_sensitive else surface.lower()
    return self._reverse.get(key)
