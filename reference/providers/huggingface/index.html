    HuggingFaceProvider¶

    wordbase.providers.huggingface ¶

    HuggingFace provider for local and hosted models.

    This module implements the LLMProvider interface for HuggingFace Transformers models. It supports both local inference and models hosted on the HuggingFace Hub, with features including device placement, quantization, and token probability extraction.

    The provider handles
    • Loading models and tokenizers from HuggingFace Hub or local paths
    • Device management (CPU, CUDA, MPS, auto)
    • 8-bit quantization for memory efficiency
    • Chat template formatting or fallback manual formatting
    • Token-level probability extraction
    • Sampling strategies (greedy, nucleus, top-k, beam search)
    • Multimodal (text + image) input for vision-language models, when constructed with multimodal=True

    Typical usage example:

    from wordbase.providers.huggingface import HuggingFaceProvider
from wordbase.types import Message

provider = HuggingFaceProvider(
    model_name="gpt2",
    device="cuda",
    load_in_8bit=True
)

messages = [Message(role="user", content="Hello, how are you?")]
completions = provider.generate(
    messages,
    temperature=0.7,
    max_tokens=100,
    return_token_probs=True
)

    def __init__(
    self,
    model_name: str,
    device: str = "auto",
    load_in_8bit: bool = False,
    multimodal: bool = False,
    **model_kwargs: Any,
):
    """Initialize HuggingFace provider.

    Args:
        model_name: HuggingFace model identifier or local path.
        device: Device placement ("auto", "cuda", "cpu", "mps").
        load_in_8bit: Use 8-bit quantization.
        multimodal: Load a vision-language model (AutoProcessor +
            AutoModelForImageTextToText) instead of a text-only model.
        **model_kwargs: Additional arguments for
            AutoModelForCausalLM.from_pretrained() (or
            AutoModelForImageTextToText.from_pretrained() when
            multimodal=True).

    Raises:
        ModelNotFoundError: If the model cannot be loaded.
    """
    super().__init__()

    self.model_name = model_name
    self.device = device
    self.load_in_8bit = load_in_8bit
    self.multimodal = multimodal
    self._processor: Any = None

    # Prepare model loading kwargs
    load_kwargs = {"device_map": device, **model_kwargs}

    if load_in_8bit:
        load_kwargs["load_in_8bit"] = True

    # MPS on macOS < 14 lacks support for several ops used during
    # generation (e.g. torch.isin on Long tensors, BFloat16). Fall back
    # to CPU in that case.
    if sys.platform == "darwin" and device in ("mps", "auto"):
        if torch.backends.mps.is_available():
            mac_ver = tuple(int(x) for x in platform.mac_ver()[0].split(".")[:2])
            if mac_ver < (14, 0):
                load_kwargs["device_map"] = "cpu"

    # Load model and tokenizer (or processor, for vision-language models)
    try:
        if multimodal:
            self._processor = AutoProcessor.from_pretrained(model_name)
            self.tokenizer = self._processor.tokenizer
            self.model = AutoModelForImageTextToText.from_pretrained(
                model_name, **load_kwargs
            )
        else:
            self.tokenizer = AutoTokenizer.from_pretrained(model_name)
            self.model = AutoModelForCausalLM.from_pretrained(
                model_name, **load_kwargs
            )
    except Exception as e:
        raise ModelNotFoundError(
            f"Failed to load model '{model_name}': {str(e)}"
        ) from e

    # Set padding token if not set (needed for batching).
    # pad_token should be different from eos_token for proper masking.
    pad_token_id = getattr(self.tokenizer, "pad_token_id", None)
    eos_token_id = getattr(self.tokenizer, "eos_token_id", None)
    if self.tokenizer.pad_token is None or pad_token_id == eos_token_id:
        # Try to use unknown token first
        unk_token = getattr(self.tokenizer, "unk_token", None)
        unk_token_id = getattr(self.tokenizer, "unk_token_id", None)
        if unk_token is not None and unk_token_id != eos_token_id:
            self.tokenizer.pad_token = unk_token
        else:
            # Add a new pad token
            self.tokenizer.add_special_tokens({"pad_token": "<|pad|>"})
            # Resize model embeddings to accommodate new token
            self.model.resize_token_embeddings(len(self.tokenizer))

    def generate(
    self,
    messages: list[Message],
    system_prompt: str | None = None,
    n: int = 1,
    max_tokens: int = 1024,
    temperature: float = 1.0,
    top_p: float = 1.0,
    top_k: int | None = None,
    num_beams: int = 1,
    stop_sequences: list[str] | None = None,
    stream: bool = False,
    use_cache: bool = True,
    return_token_probs: bool = False,
    logprob_aggregation: str = "arithmetic_mean",
    logprob_alpha: float = 1.0,
    return_tensors: bool = False,
    **kwargs: Any,
) -> list[dict[str, Any]]:
    """Generate completions given conversation history.

    Args:
        messages: Conversation history as Message objects.
        system_prompt: System prompt (prepended if provided).
        n: Number of completions to generate.
        max_tokens: Maximum tokens per completion.
        temperature: Sampling temperature (0 = greedy).
        top_p: Nucleus sampling threshold.
        top_k: Top-k sampling.
        num_beams: Beam search width (1 = no beam search).
        stop_sequences: Stop generation at these strings.
        stream: Enable streaming mode (not yet supported).
        use_cache: Whether to use cache for this request.
        return_token_probs: Whether to return token-level probabilities.
        logprob_aggregation: Method for aggregating token logprobs into
            text logprob when return_token_probs is True. Options: "sum",
            "arithmetic_mean", "scaled_arithmetic_mean". Defaults to
            "arithmetic_mean".
        logprob_alpha: Alpha parameter for "scaled_arithmetic_mean"
            aggregation. Formula: sum(logprobs) / (num_tokens ** alpha).
            When alpha=1.0, equivalent to "arithmetic_mean". Only used
            when logprob_aggregation="scaled_arithmetic_mean". Defaults
            to 1.0.
        return_tensors: When True, logprob values in the returned
            completions are PyTorch tensors with an attached computation
            graph instead of plain Python floats.  This enables standard
            gradient workflows::

                result = provider.generate(..., return_tensors=True)
                loss = -result[0]["logprob"]
                loss.backward()

            Caching is automatically disabled when return_tensors=True
            because cached plain-float values cannot carry gradients.
            HuggingFace-only; not applicable to API-based providers.
        **kwargs: Additional provider-specific parameters.

    Returns:
        List of completion dictionaries.

    Raises:
        InvalidRequestError: If messages contain image content but this
            provider was not constructed with multimodal=True.
    """
    if not self.multimodal and _has_image_content(messages):
        raise InvalidRequestError(
            "Message contains image content, but this HuggingFaceProvider "
            "was constructed without multimodal=True. Re-initialize with "
            "multimodal=True and a vision-language model to use images."
        )

    # Caching is incompatible with tensor outputs (tensors cannot be
    # serialised and deserialized while preserving the computation graph).
    cache_key = None
    if not return_tensors and use_cache and self._cache:
        cache_key = self._cache.make_key(
            messages,
            system_prompt,
            **_cache_key_params(
                {
                    "n": n,
                    "max_tokens": max_tokens,
                    "temperature": temperature,
                    "top_p": top_p,
                    "top_k": top_k,
                    "num_beams": num_beams,
                    "stop_sequences": stop_sequences,
                    "return_token_probs": return_token_probs,
                    **kwargs,
                }
            ),
        )
        cached_result = self._cache.get(cache_key)
        if cached_result is not None:
            return cached_result

    # Format messages into model inputs — route through the processor
    # when this provider was configured for vision-language models.
    if self.multimodal:
        prompt, images = self._format_multimodal_messages(
            messages, system_prompt=system_prompt
        )
        processor_kwargs: dict[str, Any] = {"text": prompt, "return_tensors": "pt"}
        if images:
            processor_kwargs["images"] = images
        inputs = self._processor(**processor_kwargs)
    else:
        prompt = self._format_messages(messages, system_prompt=system_prompt)
        # Use the callable form to get attention_mask too, so the model
        # does not warn about an unset mask.
        inputs = self.tokenizer(prompt, return_tensors="pt")

    # Move to model's device if needed
    if hasattr(self.model, "device"):
        inputs = {k: v.to(self.model.device) for k, v in inputs.items()}

    # Prepare generation kwargs
    gen_kwargs = self._prepare_generation_kwargs(
        n,
        max_tokens,
        temperature,
        top_p,
        top_k,
        num_beams,
        stop_sequences,
        return_token_probs,
        return_tensors=return_tensors,
    )
    gen_kwargs.update(kwargs)  # Add any additional kwargs

    # Generate
    # transformers' typed model union does not satisfy the self-type
    # of its own generation-mixin methods.
    outputs = self.model.generate(**inputs, **gen_kwargs)  # type: ignore[misc]

    # Extract and format completions
    input_length = int(inputs["input_ids"].shape[1])
    completions = self._extract_completions_from_outputs(
        outputs,
        input_length,
        n,
        return_token_probs,
        num_requests=1,
        logprob_aggregation=logprob_aggregation,
        logprob_alpha=logprob_alpha,
        return_tensors=return_tensors,
    )[
        0
    ]  # Get the first (and only) request's completions

    # Store in cache if enabled (never when return_tensors=True)
    if not return_tensors and use_cache and self._cache and cache_key:
        self._cache.set(cache_key, completions)

    return completions

    def generate_batch(
    self, batch: list[dict[str, Any]], **kwargs: Any
) -> list[list[dict[str, Any]]]:
    """Generate completions for multiple requests in batch.

    This method processes multiple generation requests efficiently using
    true batching with padding for local HuggingFace models. Requests are
    grouped by their generation parameters, and each group is processed
    as a sub-batch to maximize batching efficiency. Requests containing
    image content are routed through generate() individually instead —
    padded true-batching across images with a varying image count per
    request is significantly more fragile, so this trades some
    throughput for correctness when multimodal=True.

    Args:
        batch: List of request dictionaries, each containing:
            - messages: Conversation history (required)
            - system_prompt: Optional system prompt
            - Other generation parameters (n, max_tokens, temperature,
              return_token_probs, logprob_aggregation,
              logprob_alpha, etc.)
        **kwargs: Common parameters applied to all requests (e.g.,
            temperature, return_token_probs, logprob_aggregation,
            logprob_alpha).

    Returns:
        List of completion lists (one list per request).
    """
    if not batch:
        return []

    # Check cache for all requests
    results, uncached_requests = self._check_cache_for_batch(batch, **kwargs)

    if not uncached_requests:
        return cast(list[list[dict[str, Any]]], results)

    # Multimodal requests bypass padded true-batching and go through
    # generate() one at a time; everything else keeps the efficient
    # grouped sub-batch path.
    text_requests = []
    multimodal_requests = []
    for item in uncached_requests:
        _, request, _ = item
        if self.multimodal and _has_image_content(request.get("messages", [])):
            multimodal_requests.append(item)
        else:
            text_requests.append(item)

    if text_requests:
        param_groups = self._group_requests_by_params(text_requests)
        for param_key, group_requests in param_groups.items():
            group_results = self._process_sub_batch(
                group_requests, param_key, **kwargs
            )
            for orig_idx, completions in group_results:
                results[orig_idx] = completions

    for orig_idx, request, merged_params in multimodal_requests:
        messages = request.get("messages", [])
        system_prompt = request.get("system_prompt")
        results[orig_idx] = self.generate(
            messages, system_prompt=system_prompt, **merged_params
        )

    return cast(list[list[dict[str, Any]]], results)

    """Score continuation strings via teacher-forcing log-probabilities.

    Appends each continuation to the prompt and runs a batched forward
    pass to obtain the log probability of each continuation token given
    its prefix.  The score for a continuation is the sum of its per-token
    log probabilities.

    Args:

        continuations: Continuation strings to score.
        return_tensors: When True, return each score as a scalar
            ``torch.Tensor`` with an attached computation graph so

            return plain Python floats.

    Returns:
        Dict mapping each continuation string to its log-probability
        sum.  Values are ``float`` when ``return_tensors=False``, or
        scalar ``torch.Tensor`` with ``grad_fn`` when
        ``return_tensors=True``.

    Warns:
        MultiTokenSurfaceWarning: When any continuation tokenizes to
            more than one token.  Teacher-forcing is still applied, but
            multi-token scoring may be slower.
    """
    import torch.nn.functional as F  # pylint: disable=import-outside-toplevel

    if not continuations:
        return {}

    # Tokenize the prompt
    prompt_inputs = self.tokenizer(prompt, return_tensors="pt")
    prompt_ids = prompt_inputs["input_ids"][0]  # (prompt_len,)
    prompt_len = int(prompt_ids.shape[0])

    # Tokenize each continuation; warn on multi-token surface forms
    cont_ids_list = []
    for cont in continuations:
        ids = self.tokenizer.encode(cont, add_special_tokens=False)
        if len(ids) > 1:
            warnings.warn(
                f"Surface form {cont!r} tokenizes to {len(ids)} tokens. "
                "Teacher-forcing is used to score multi-token continuations.",
                MultiTokenSurfaceWarning,
                stacklevel=2,
            )
        cont_ids_list.append(ids)

    max_cont_len = max((len(ids) for ids in cont_ids_list), default=0)
    if max_cont_len == 0:
        # All continuations are empty — nothing to score
        val: Any = torch.tensor(0.0) if return_tensors else 0.0
        return {cont: val for cont in continuations}

    max_seq_len = prompt_len + max_cont_len
    pad_id = self.tokenizer.pad_token_id

    # Build padded input batch
    batch_input_ids = []
    batch_attn_mask = []
    for cont_ids in cont_ids_list:
        real_len = prompt_len + len(cont_ids)
        padding = max_seq_len - real_len
        row = prompt_ids.tolist() + list(cont_ids) + [pad_id] * padding
        mask = [1] * real_len + [0] * padding
        batch_input_ids.append(row)
        batch_attn_mask.append(mask)

    device = (
        self.model.device if hasattr(self.model, "device") else torch.device("cpu")
    )
    input_ids_t = torch.tensor(batch_input_ids, dtype=torch.long, device=device)
    attn_mask_t = torch.tensor(batch_attn_mask, dtype=torch.long, device=device)

    if return_tensors:
        embedding_layer = self.model.get_input_embeddings()
        input_embeds = embedding_layer(input_ids_t).detach().requires_grad_(True)
        fwd_out = self.model(inputs_embeds=input_embeds, attention_mask=attn_mask_t)
    else:
        with torch.no_grad():
            fwd_out = self.model(input_ids=input_ids_t, attention_mask=attn_mask_t)

    logits = fwd_out.logits  # (num_conts, max_seq_len, vocab_size)
    log_probs = F.log_softmax(logits, dim=-1)

    result: dict[str, Any] = {}
    for i, (cont, cont_ids) in enumerate(
        zip(continuations, cont_ids_list, strict=True)
    ):
        if not cont_ids:
            result[cont] = torch.tensor(0.0) if return_tensors else 0.0
            continue

        cont_len = len(cont_ids)
        # Logit at position (prompt_len - 1 + j) predicts continuation
        # token j (0-indexed).
        relevant = log_probs[
            i, prompt_len - 1 : prompt_len - 1 + cont_len, :
        ]  # (cont_len, vocab_size)
        cont_token_ids = torch.tensor(cont_ids, dtype=torch.long, device=device)
        token_lps = relevant.gather(1, cont_token_ids.unsqueeze(1)).squeeze(1)
        score = token_lps.sum()
        result[cont] = score if return_tensors else float(score.item())

    return result

    def first_token_logprobs(
    self,
    prompt: str,
    *,
    return_tensors: bool = False,
) -> dict[int, Any]:
    """Get the next-token log-probability distribution for a prompt.

    Runs a single forward pass (no generation) and returns the full
    vocabulary log-probability distribution at the final prompt position.

    Args:
        prompt: The prompt string to condition on.
        return_tensors: When True, return each value as a scalar
            ``torch.Tensor`` with an attached computation graph.  When
            False (default), return plain Python floats.

    Returns:
        Dict mapping token ID (``int``) to its log-probability.  Values
        are ``float`` when ``return_tensors=False``, or scalar
        ``torch.Tensor`` with ``grad_fn`` when ``return_tensors=True``.
    """
    import torch.nn.functional as F  # pylint: disable=import-outside-toplevel

    inputs = self.tokenizer(prompt, return_tensors="pt")
    device = (
        self.model.device if hasattr(self.model, "device") else torch.device("cpu")
    )
    inputs = {k: v.to(device) for k, v in inputs.items()}

    if return_tensors:
        embedding_layer = self.model.get_input_embeddings()
        input_embeds = (
            embedding_layer(inputs["input_ids"]).detach().requires_grad_(True)
        )
        fwd_out = self.model(inputs_embeds=input_embeds)
    else:
        with torch.no_grad():
            fwd_out = self.model(**inputs)

    logits = fwd_out.logits  # (1, seq_len, vocab_size)
    last_logits = logits[0, -1, :]  # (vocab_size,)
    log_probs = F.log_softmax(last_logits, dim=-1)  # (vocab_size,)

    vocab_size = int(log_probs.shape[0])
    if return_tensors:
        return {i: log_probs[i] for i in range(vocab_size)}
    lp_list: list[float] = log_probs.tolist()
    return {i: lp for i, lp in enumerate(lp_list)}
