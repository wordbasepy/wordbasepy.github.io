    VLLMProvider¶

    wordbase.providers.vllm_provider ¶

    vLLM provider for high-throughput model inference.

    This module implements the LLMProvider interface using vLLM, an open-source library that delivers high-throughput and memory-efficient LLM inference via PagedAttention for KV-cache management and continuous batching.

    The provider handles
    • Loading models via the vLLM LLM engine
    • Chat template formatting with manual fallback
    • Sampling strategies (temperature, nucleus, top-k)
    • Per-token log probability extraction via SamplingParams.logprobs
    • Batch processing via vLLM's built-in continuous batching engine
    • Response caching

    Typical usage example:

    from wordbase.providers.vllm_provider import VLLMProvider
from wordbase.types import Message

provider = VLLMProvider(
    model_name="meta-llama/Llama-2-7b-chat-hf",
    tensor_parallel_size=2,
)
messages = [Message(role="user", content="Hello, how are you?")]
completions = provider.generate(
    messages, temperature=0.7, max_tokens=100, return_token_probs=True
)

    def __init__(self, model_name: str, **engine_kwargs: Any):
    """Initialize the vLLM provider.

    Args:
        model_name: HuggingFace model identifier or local path.
        **engine_kwargs: Additional arguments forwarded to vllm.LLM().

    Raises:
        ModelNotFoundError: If the model cannot be loaded.
    """
    super().__init__()
    self.model_name = model_name

    try:
        self.llm = LLM(model=model_name, **engine_kwargs)
    except Exception as e:
        raise ModelNotFoundError(
            f"Failed to load model '{model_name}': {str(e)}"
        ) from e

    def generate(
    self,
    messages: list[Message],
    system_prompt: str | None = None,
    n: int = 1,
    max_tokens: int = 1024,
    temperature: float = 1.0,
    top_p: float = 1.0,
    top_k: int | None = None,
    num_beams: int = 1,
    stop_sequences: list[str] | None = None,
    stream: bool = False,
    use_cache: bool = True,
    return_token_probs: bool = False,
    logprob_aggregation: str = "arithmetic_mean",
    logprob_alpha: float = 1.0,
    **kwargs: Any,
) -> list[dict[str, Any]]:
    """Generate completions given conversation history.

    Streaming (stream=True) is not yet implemented and is ignored.

    Args:
        messages: Conversation history as Message objects.
        system_prompt: System prompt prepended to the conversation.
        n: Number of completions to generate.
        max_tokens: Maximum tokens per completion.
        temperature: Sampling temperature (0.0 = greedy decoding).
        top_p: Nucleus sampling cumulative probability threshold.
        top_k: Top-k token filtering (None disables it).
        num_beams: Must be 1; vLLM does not support beam search.
        stop_sequences: Strings that stop generation when encountered.
        stream: Ignored; streaming is not yet implemented.
        use_cache: Whether to use the configured cache.
        return_token_probs: Whether to return per-token log probabilities.
        logprob_aggregation: How to aggregate token logprobs into a single
            text logprob. Options: "sum", "arithmetic_mean",
            "scaled_arithmetic_mean".
        logprob_alpha: Alpha for "scaled_arithmetic_mean".
            Formula: sum(logprobs) / (n_tokens ** alpha).
        **kwargs: Ignored additional parameters for interface compatibility.

    Returns:
        List of n completion dictionaries, each with:
            - text: Generated text string.
            - tokens: Token dicts with logprobs (if return_token_probs).
            - finish_reason: "stop" or "length".
            - metadata: Model name and output_tokens count.
            - logprob: Aggregated log probability (if return_token_probs).

    Raises:
        InvalidRequestError: If num_beams > 1 (vLLM does not support beam search).
    """
    if num_beams > 1:
        raise InvalidRequestError(
            "vLLM does not support beam search. Use num_beams=1 "
            "or switch to HuggingFaceProvider for beam search."
        )
    cache_key = None
    if use_cache and self._cache:
        cache_key = self._cache.make_key(
            messages,
            system_prompt,
            **_cache_key_params(
                {
                    "n": n,
                    "max_tokens": max_tokens,
                    "temperature": temperature,
                    "top_p": top_p,
                    "top_k": top_k,
                    "stop_sequences": stop_sequences,
                    "return_token_probs": return_token_probs,
                    **kwargs,
                }
            ),
        )
        cached = self._cache.get(cache_key)
        if cached is not None:
            return cached

    prompt = self._format_messages(messages, system_prompt=system_prompt)
    sampling_params = self._build_sampling_params(
        n,
        max_tokens,
        temperature,
        top_p,
        top_k,
        stop_sequences,
        return_token_probs,
    )
    request_outputs = self.llm.generate([prompt], sampling_params)

    completions = self._format_completions(
        request_outputs,
        return_token_probs,
        logprob_aggregation,
        logprob_alpha,
    )[0]

    if use_cache and self._cache and cache_key is not None:
        self._cache.set(cache_key, completions)

    return completions

    def generate_batch(
    self, batch: list[dict[str, Any]], **kwargs: Any
) -> list[list[dict[str, Any]]]:
    """Generate completions for multiple requests using vLLM batching.

    Requests are grouped by their generation parameters and submitted
    together to vLLM's continuous batching engine, which handles
    scheduling and KV-cache management internally.

    Args:
        batch: List of request dictionaries, each optionally containing:
            - messages: Conversation history (required).
            - system_prompt: Optional system prompt.
            - n, max_tokens, temperature, top_p, top_k,
              stop_sequences, return_token_probs, logprob_aggregation,
              logprob_alpha: Per-request generation parameters.
        **kwargs: Common parameters applied to all requests (overridden
            by per-request values with the same key).

    Returns:
        List of completion lists, one inner list per batch request.
    """
    if not batch:
        return []

    results: list[Any] = [None] * len(batch)
    uncached_requests = []

    for idx, request in enumerate(batch):
        messages = request.get("messages", [])
        system_prompt = request.get("system_prompt")
        use_cache = request.get("use_cache", kwargs.get("use_cache", True))

        merged_params: dict[str, Any] = {**kwargs, **request}
        merged_params.pop("messages", None)
        merged_params.pop("system_prompt", None)

        if use_cache and self._cache:
            cache_key = self._cache.make_key(
                messages, system_prompt, **_cache_key_params(merged_params)
            )
            cached = self._cache.get(cache_key)
            if cached is not None:
                results[idx] = cached
                continue

        uncached_requests.append((idx, request, merged_params))

    if not uncached_requests:
        return results

    # Group uncached requests by generation parameters so that requests
    # sharing identical settings are dispatched to vLLM together.
    param_groups: dict[tuple, list[Any]] = {}
    for idx, request, merged_params in uncached_requests:
        param_key = (
            merged_params.get("n", 1),
            merged_params.get("max_tokens", 1024),
            merged_params.get("temperature", 1.0),
            merged_params.get("top_p", 1.0),
            merged_params.get("top_k"),
            tuple(merged_params.get("stop_sequences", []) or []),
            merged_params.get("return_token_probs", False),
            merged_params.get("logprob_aggregation", "arithmetic_mean"),
            merged_params.get("logprob_alpha", 1.0),
        )
        if param_key not in param_groups:
            param_groups[param_key] = []
        param_groups[param_key].append((idx, request, merged_params))

    for param_key, group_requests in param_groups.items():
        (
            n,
            max_tokens,
            temperature,
            top_p,
            top_k,
            stop_sequences_tuple,
            return_token_probs,
            logprob_aggregation,
            logprob_alpha,
        ) = param_key
        stop_sequences = (
            list(stop_sequences_tuple) if stop_sequences_tuple else None
        )

        prompts = [
            self._format_messages(
                req.get("messages", []),
                system_prompt=req.get("system_prompt"),
            )
            for _, req, _ in group_requests
        ]
        sampling_params = self._build_sampling_params(
            n,
            max_tokens,
            temperature,
            top_p,
            top_k,
            stop_sequences,
            return_token_probs,
        )
        request_outputs = self.llm.generate(prompts, sampling_params)
        group_completions = self._format_completions(
            request_outputs,
            return_token_probs,
            logprob_aggregation,
            logprob_alpha,
        )

        for batch_idx, (orig_idx, request, merged_params) in enumerate(
            group_requests
        ):
            completions = group_completions[batch_idx]

            use_cache_req = request.get("use_cache", kwargs.get("use_cache", True))
            if use_cache_req and self._cache:
                req_msgs = request.get("messages", [])
                req_sys = request.get("system_prompt")
                cache_key = self._cache.make_key(
                    req_msgs, req_sys, **_cache_key_params(merged_params)
                )
                self._cache.set(cache_key, completions)

            results[orig_idx] = completions

    return results

    def count_tokens(self, text: str) -> int:
    """Count tokens in text using the model's tokenizer.

    Args:
        text: Text to tokenize and count.

    Returns:
        Number of tokens in the text.
    """
    tokenizer = self._get_tokenizer()
    return len(tokenizer.encode(text))
