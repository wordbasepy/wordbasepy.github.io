    Calibration¶

    wordbase.calibration ¶

    Calibration of label log-probability distributions.

    Raw label log-probabilities produced by aggregate_logprobs/aggregate_counts are often over- or under-confident relative to the true label frequencies. This module provides:

    • CalibrationConfig: an immutable description of a fitted calibration transform (temperature scaling or Platt scaling).
    • apply_calibration: applies a CalibrationConfig to a label log-probability dict.
    • CalibrationFitter: fits a CalibrationConfig from held-out predictions and their true labels, optionally weighting each example by the effective sample size of its SampleUncertainty.

    Typical usage example:

    from wordbase.calibration import CalibrationFitter, apply_calibration

fitter = CalibrationFitter(method="temperature")
config = fitter.fit(predictions, true_labels)
calibrated = apply_calibration(predictions[0], config)

    def fit(
    self,
    predictions: list[dict[T, float]],
    labels: list[T],
    *,
    uncertainties: list[SampleUncertainty[T] | None] | None = None,
) -> CalibrationConfig:
    """Fit a CalibrationConfig to minimise held-out negative log-likelihood.

    Args:
        predictions: One label log-probability dict per example, as
            produced by aggregate_logprobs/aggregate_counts.
        labels: The true label for each example, same length and order
            as predictions.
        uncertainties: Optional per-example SampleUncertainty objects
            (e.g. from aggregate_counts). When provided, each example is
            weighted by its uncertainty.n_samples, so low-variance
            estimates influence the fit more than high-variance ones.

    Returns:
        A fitted CalibrationConfig for self.method.

    Raises:
        ValueError: If predictions and labels have different lengths, or
            if uncertainties is provided with a different length.
    """
    if len(predictions) != len(labels):
        raise ValueError(
            "predictions and labels must have the same length "
            f"({len(predictions)} != {len(labels)})."
        )
    if uncertainties is not None and len(uncertainties) != len(predictions):
        raise ValueError(
            "uncertainties must have the same length as predictions "
            f"({len(uncertainties)} != {len(predictions)})."
        )

    weights = _resolve_weights(len(predictions), uncertainties)

    if self.method == "temperature":
        temperature = self._fit_temperature(predictions, labels, weights)
        return CalibrationConfig(method="temperature", temperature=temperature)

    platt_a, platt_b = self._fit_platt(predictions, labels, weights)
    return CalibrationConfig(method="platt", platt_a=platt_a, platt_b=platt_b)

    def apply_calibration[T](
    logprobs: Mapping[T, float | torch.Tensor],  # noqa: F821
    config: CalibrationConfig,
) -> dict[T, float | torch.Tensor]:  # noqa: F821
    """Apply a calibration transform to a label log-probability dict.

    Values may be plain floats or torch.Tensor instances; the latter keep
    their computation graph intact through the transform and the
    renormalisation step (so gradients flow through when called with
    tensors produced by aggregate_logprobs).

    Args:
        logprobs: Dict mapping each label to its (normalised) log-probability,
            as produced by aggregate_logprobs/aggregate_counts.
        config: The calibration transform to apply.

    Returns:
        A dict with the same keys as logprobs, containing calibrated and
        renormalised log-probabilities. Returned unchanged when
        config.method == "none".

    Raises:
        CalibrationNotFittedError: If config.method is not "none" but the
            parameters required for that method have not been set.
    """
    if config.method == "none":
        return dict(logprobs)

    if config.method == "temperature":
        if config.temperature is None:
            raise CalibrationNotFittedError(
                "CalibrationConfig.temperature is None; call "
                "CalibrationFitter.fit() before applying temperature "
                "calibration."
            )
        scaled = {label: lp / config.temperature for label, lp in logprobs.items()}
        return _renormalize(scaled)

    if config.platt_a is None or config.platt_b is None:
        raise CalibrationNotFittedError(
            "CalibrationConfig.platt_a/platt_b are None; call "
            "CalibrationFitter.fit() before applying Platt calibration."
        )
    scaled = {
        label: config.platt_a * lp + config.platt_b for label, lp in logprobs.items()
    }
    return _renormalize(scaled)
